    library(data.table)

    #library(randomForest)

    library(autosql)

    # -------------------------------------------------

    set.autosql.project("examples")

    # -------------------------------------------------
    # Create artificial data set - don't worry. You
    # do not have to understand this part.

    peripheral.table = data.table(
        column_01=runif(250000),
        join_key=floor(runif(250000, 0, 500)),
        time_stamp=runif(250000)
    )

    temp.table = data.table(
        column_01=runif(500),
        join_key=(0:499),
        time_stamp=runif(500)
    )

    temp.table = merge(
        peripheral.table,
        temp.table,
        by="join_key"
    )

    temp.table = temp.table[time_stamp.x < time_stamp.y & time_stamp.x > time_stamp.y - 0.5,]

    population.table = temp.table[, .(
        targets=.N, 
        column_01=mean(column_01.y),
        time_stamp=mean(time_stamp.y)
        ), 
        by=join_key
    ]

    rm(temp.table)

    # -------------------------------------------------
    # Define fit and predict function - you 
    # only have to do this, if you are not using the 
    # built-in XGBoost library.

    # The "fit" function takes the features generated
    # by model.transform(...) and returns the fitted
    # predictor.
    autosql.fit <- function( features ) { 

        rf <- randomForest( 
        targets ~ ., 
        data=features,
        ntree=50
        )

        rf
    }

    # The "predict" function takes the predictor
    # fitted by the "fit" function as well as the
    # features and returns the predictions.
    autosql.predict <- function( predictor, features ) { 

        predictions <- predict(
        object=predictor, 
        newdata=features
        )

        predictions
    }

    # -------------------------------------------------
    # Build data model

    population.placeholder <- create.autosql.placeholder(
        "POPULATION",
        join.keys=c( "join_key" ),
        time.stamps=c( "time_stamp" ),
        targets=c( "targets" )
    )

    peripheral.placeholder <- create.autosql.placeholder(
        "PERIPHERAL",
        join.keys=c( "join_key" ),
        time.stamps=c( "time_stamp" )
    )

    population.placeholder <- join.autosql.placeholder(
        population.placeholder,
        peripheral.placeholder,
        "join_key",
        "time_stamp"
    )

    model = create.autosql.model(
        population=population.placeholder,
        peripheral=list(peripheral.placeholder),
        aggregation=list(        
        autosql.count, 
        autosql.sum 
        ),
        use.timestamps=TRUE,
        num.features=10,
        max.length=1,
        fast.training=FALSE,
        min.num.samples=200,
        shrinkage=0.0,
        grid.factor=1.0,
        share.aggregations=1.0,
        feature.selector=autosql.create.xgboost.regressor(),
        predictor=autosql.create.xgboost.regressor()#,
        #fit=autosql.fit,
        #predict=autosql.predict,
    )

    send.autosql.model(model)

    # ----------------
    # Fit model

    model = fit.autosql.model(
        model,
        population.table,
        peripheral.table
    )

    scores = score.autosql.model(
        model,
        population.table,
        peripheral.table
    )

    features = autosql.model.transform(
        model,
        population.table,
        peripheral.table
    )

    predictions = autosql.model.predict(
        model,
        population.table,
        peripheral.table
    )

    cat( autosql.model.to.sql(model) )

    # ----------------
    # Plot results

    plot( 
        features[,1], 
        features[,"targets"], 
        type="p",
        col="blue",
        xlab="predictions",
        ylab="targets"
    )

    plot( 
        predictions, 
        features[,"targets"], 
        type="p",
        col="blue",
        xlab="predictions",
        ylab="targets"
    )
