    library(data.table)

    library(autosql)

    # -------------------------------------------------

    set.autosql.project("examples")

    # -------------------------------------------------
    # Create artificial data set - don't worry. You
    # do not have to understand this part.

    peripheral.table = data.table(
        column_01=as.character(floor(runif(250000, 1, 10))),
        join_key=floor(runif(250000, 0, 500)),
        time_stamp=runif(250000)
    )

    temp.table = data.table(
        column_01=as.character(floor(runif(500, 1, 10))),
        join_key=(0:499),
        time_stamp=runif(500)
    )

    temp.table = merge(
        peripheral.table,
        temp.table,
        by="join_key"
    )

    temp.table = temp.table[time_stamp.x < time_stamp.y & column_01.x == column_01.y,]

    population.table = temp.table[, .(
        targets=.N, 
        column_01=max(column_01.y),
        time_stamp=mean(time_stamp.y)
        ), 
        by=join_key
    ]

    rm(temp.table)

    #----------------
    # Set up units

    units = list()

    units[["column_01"]] = "unit_01"

    # -------------------------------------------------
    # Build data model

    population.placeholder <- create.autosql.placeholder(
        "POPULATION",
        categorical=c( "column_01" ),
        join.keys=c( "join_key" ),
        time.stamps=c( "time_stamp" ),
        targets=c( "targets" )
    )

    peripheral.placeholder <- create.autosql.placeholder(
        "PERIPHERAL",
        categorical=c( "column_01" ),
        join.keys=c( "join_key" ),
        time.stamps=c( "time_stamp" )
    )

    population.placeholder <- join.autosql.placeholder(
        population.placeholder,
        peripheral.placeholder,
        "join_key",
        "time_stamp"
    )

    model = create.autosql.model(
        name="Model 1",
        population=population.placeholder,
        peripheral=list(peripheral.placeholder),
        aggregation=list(        
        autosql.count, 
        autosql.sum,
        autosql.min
        ),
        units=units, # passing the units to the model
        use.timestamps=TRUE,
        num.features=10,
        max.length=2,
        fast.training=FALSE,
        min.num.samples=200,
        shrinkage=0.0,
        grid.factor=1.0,
        share.aggregations=1.0,
        num.threads=6,
        regularization=0.01,
        predictor=autosql.create.xgboost.regressor()
    )

    send.autosql.model(model)

    # ----------------
    # Fit model

    model = fit.autosql.model(
        model,
        population.table,
        peripheral.table
    )

    features = autosql.model.transform(
        model,
        population.table,
        peripheral.table
    )

    cat( autosql.model.to.sql(model) )

    # ----------------
    # Plot results

    plot( 
        features[,1], 
        population.table[["targets"]],
        type="p",
        col="blue",
        xlab="predictions",
        ylab="targets"
    )

    pred = rowMeans( features )

    plot( 
        pred, 
        population.table[["targets"]],
        type="p",
        col="blue",
        xlab="predictions",
        ylab="targets"
    )
